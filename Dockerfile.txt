# 1. Base image with Python 3.12
FROM python:3.12-slim

# 2. Set working directory
WORKDIR /app

# 3. Install system dependencies (Java, Git, etc. for PySpark & AutoGluon)
RUN apt-get update && apt-get install -y --no-install-recommends \
    openjdk-17-jdk \
    git \
    curl \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

# 4. Set JAVA_HOME for PySpark
ENV JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64
ENV PATH=$JAVA_HOME/bin:$PATH

# 5. Copy requirements.txt and install dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 6. Install AutoGluon separately (to avoid build errors)
RUN pip install --no-cache-dir autogluon==1.4.0

# 7. Copy source code and models
COPY src/ ./src
COPY models/ ./models
COPY data/ ./data

# 8. Expose API (FastAPI)
EXPOSE 8000

# 9. Expose MLflow UI (optional)
EXPOSE 5000

# 10. Default command: run FastAPI API
CMD ["uvicorn", "src.api:app", "--host", "0.0.0.0", "--port", "8000"]
